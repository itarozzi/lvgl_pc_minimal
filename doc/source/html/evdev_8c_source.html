60/*
61 * Initialize the evdev driver
62 *
63 * @param backend the backend descriptor
64 */
65int backend_init_evdev(backend_t * backend)
66{
67 LV_ASSERT_NULL(backend);
68 backend->handle->indev = malloc(sizeof(indev_backend_t));
69 LV_ASSERT_NULL(backend->handle->indev);
70
71 backend->handle->indev->init_indev = init_pointer_evdev;
72
73 backend->name = backend_name;
74 backend->type = BACKEND_INDEV;
75 return 0;
76}

98/*
99 * Set cursor icon
100 *
101 * @description Once the input device is discovered set the mouse cursor icon
102 * @note called by the LVGL evdev driver
103 * @param indev the input device
104 * @param type the type of the input device
105 * @param user_data the user data
106 */
107static void discovery_cb(lv_indev_t * indev, lv_evdev_type_t type, void * user_data)
108{
109 LV_LOG_USER("new '%s' device discovered", type == LV_EVDEV_TYPE_REL ? "REL" :
110 type == LV_EVDEV_TYPE_ABS ? "ABS" :
111 type == LV_EVDEV_TYPE_KEY ? "KEY" :
112 "unknown");
113
114 lv_display_t * disp = user_data;
115 lv_indev_set_display(indev, disp);
116
117 if(type == LV_EVDEV_TYPE_REL) {
118 set_mouse_cursor_icon(indev, disp);
119 }
120}

122/*
123 * Set cursor icon
124 *
125 * @description Enables a pointer (touchscreen/mouse) input device
126 * @param display the display on which to create
127 * @param indev the input device to set the cursor on
128 */
129static void set_mouse_cursor_icon(lv_indev_t * indev, lv_display_t * display)
130{
131 /* Set the cursor icon */
132 LV_IMAGE_DECLARE(mouse_cursor_icon);
133 lv_obj_t * cursor_obj = lv_image_create(lv_display_get_screen_active(display));
134 lv_image_set_src(cursor_obj, &mouse_cursor_icon);
135 lv_indev_set_cursor(indev, cursor_obj);
136
137 /* delete the mouse cursor icon if the device is removed */
138 lv_indev_add_event_cb(indev, indev_deleted_cb, LV_EVENT_DELETE, cursor_obj);
139
140}

142/*
143 * Initialize a mouse pointer device
144 *
145 * Enables a pointer (touchscreen/mouse) input device
146 * Use 'evtest' to find the correct input device. /dev/input/by-id/ is recommended if possible
147 * Use /dev/input/by-id/my-mouse-or-touchscreen or /dev/input/eventX
148 *
149 * If LV_LINUX_EVDEV_POINTER_DEVICE is not set, automatic evdev discovery will start
150 *
151 * @param display the LVGL display
152 *
153 * @return input device
154 */
155static lv_indev_t * init_pointer_evdev(lv_display_t * display)
156{
157 const char * input_device = getenv("LV_LINUX_EVDEV_POINTER_DEVICE");
158
159 if(input_device == NULL) {
160 LV_LOG_USER("Using evdev automatic discovery.");
161 lv_evdev_discovery_start(discovery_cb, display);
162 return NULL;
163 }
164
165 lv_indev_t * indev = lv_evdev_create(LV_INDEV_TYPE_POINTER, input_device);
166
167 if(indev == NULL) {
168 return NULL;
169 }
170
171 lv_indev_set_display(indev, display);
172
173 set_mouse_cursor_icon(indev, display);
174 return indev;
175}
