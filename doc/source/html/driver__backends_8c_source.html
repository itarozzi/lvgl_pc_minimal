113void driver_backends_register(void)
114{
115 int i;
116 backend_init_t init_backend;
117 backend_t * b;
118
119 i = 0;
120 if(backends[i] != NULL) {
121 /* backends are already registered - leave */
122 return;
123 }
124
125 while((init_backend = available_backends[i]) != NULL) {
126
127 b = malloc(sizeof(backend_t));
128 LV_ASSERT_NULL(b);
129
130 b->handle = malloc(sizeof(backend_handle_t));
131
132 init_backend = available_backends[i];
133 LV_ASSERT_NULL(init_backend);
134
135 init_backend(b);
136 backends[i] = b;
137 i++;
138 }
139}

141int driver_backends_init_backend(char * backend_name)
142{
143 backend_t * b;
144 int i;
145 display_backend_t * dispb;
146 indev_backend_t * indevb;
147
148 if(backends[0] == NULL) {
149 LV_LOG_ERROR("Please call driver_backends_register first");
150 return -1;
151 }
152
153 if(backend_name == NULL) {
154
155 /*
156 * Set default display backend - which is the first defined
157 * item in available_backends array
158 */
159 LV_ASSERT_NULL(backends[0]);
160 b = backends[0];
161
162 if(b->type != BACKEND_DISPLAY) {
163 LV_LOG_ERROR("The default backend: %s is not a display driver backend", b->name);
164 return -1;
165 }
166
167 backend_name = backends[0]->name;
168 }
169
170 i = 0;
171 while((b = backends[i]) != NULL) {
172
173 /* Check if such a backend exists */
174 if(strcmp(b->name, backend_name) == 0) {
175
176 if(b->type == BACKEND_DISPLAY) {
177 /* Initialize the display */
178
179 dispb = b->handle->display;
180 LV_ASSERT_NULL(dispb->init_display);
181 dispb->display = dispb->init_display();
182
183 if(dispb->display == NULL) {
184 LV_LOG_ERROR("Failed to init display with %s backend", b->name);
185 return -1;
186 }
187
188 sel_display_backend = b;
189 LV_LOG_INFO("Initialized %s display backend", b->name);
190 break;
191
192 }
193 else if(b->type == BACKEND_INDEV) {
194 /* Initialize input device */
195
196 indevb = b->handle->indev;
197 LV_ASSERT_NULL(indevb->init_indev);
198
199 /* The display driver backend - has to be initialized first */
200 if(sel_display_backend == NULL) {
201 LV_LOG_ERROR(
202 "Failed to init indev backend: %s - display needs to be initialized",
203 b->name);
204 return -1;
205 }
206
207 LV_LOG_INFO("Initialized %s indev backend", b->name);
208
209 dispb = sel_display_backend->handle->display;
210
211 LV_ASSERT_NULL(dispb->display);
212 indevb->init_indev(dispb->display);
213 break;
214 }
215 }
216 i++;
217 }
218
219 return 0;
220}

247int driver_backends_is_supported(char * backend_name)
248{
249 char c;
250 backend_t * b;
251 char * name = backend_name;
252 int i = 0;
253
254 while((c = *backend_name) != '\0') {
255 *backend_name = toupper(c);
256 backend_name++;
257 }
258
259 while((b = backends[i++]) != NULL) {
260 if(strcmp(b->name, name) == 0) {
261 return 1;
262 }
263 }
264
