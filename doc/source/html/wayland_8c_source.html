89static lv_display_t * init_wayland(void)
90{

92 lv_group_t * g;
93
94 disp = lv_wayland_window_create(settings.window_width, settings.window_height,
95 "LVGL Simulator", NULL);
96
97 if(disp == NULL) {
98 die("Failed to initialize Wayland backend\n");
99 }
100
101 if(settings.fullscreen) {
102 lv_wayland_window_set_fullscreen(disp, true);
103 }
104 else if(settings.maximize) {
105 lv_wayland_window_set_maximized(disp, true);
106 }
107
108 g = lv_group_create();
109 lv_group_set_default(g);
110 lv_indev_set_group(lv_wayland_get_keyboard(disp), g);
111 lv_indev_set_group(lv_wayland_get_pointeraxis(disp), g);
112
113 return disp;

123static void run_loop_wayland(void)
124{
125
126 uint32_t idle_time;
127
128 /* Handle LVGL tasks */

133 if(idle_time != 0) {
134 usleep(idle_time * 1000);
135 }
