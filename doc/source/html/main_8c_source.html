116static void configure_simulator(int argc, char ** argv)
117{
118 int opt = 0;
119
120 selected_backend = NULL;
121 driver_backends_register();
122
123 const char * env_w = getenv("LV_SIM_WINDOW_WIDTH");
124 const char * env_h = getenv("LV_SIM_WINDOW_HEIGHT");
125 /* Default values */
126 settings.window_width = atoi(env_w ? env_w : DEFAULT_DISPLAY_W);
127 settings.window_height = atoi(env_h ? env_h : DEFAULT_DISPLAY_H);
128
129 /* Parse the command-line options. */
130 while((opt = getopt(argc, argv, "b:fmW:H:R:BVh")) != -1) {
131 switch(opt) {
132 case 'h':
133 print_usage();
134 exit(EXIT_SUCCESS);
135 break;
136 case 'V':
137 print_lvgl_version();
138 exit(EXIT_SUCCESS);
139 break;
140 case 'B':
141 driver_backends_print_supported();
142 exit(EXIT_SUCCESS);
143 break;
144 case 'b':
145 if(driver_backends_is_supported(optarg) == 0) {
146 die("error no such backend: %s\n", optarg);
147 }
148 selected_backend = strdup(optarg);
149 break;
150 case 'f':
151 settings.fullscreen = true;
152 break;
153 case 'm':
154 settings.maximize = true;
155 break;
156 case 'W':
157 settings.window_width = atoi(optarg);
158 break;
159 case 'H':
160 settings.window_height = atoi(optarg);
161 break;
162 case 'R':
163 switch(atoi(optarg)) {
164 case 0:
165 settings.rotation = LV_DISPLAY_ROTATION_0;
166 break;
167 case 90:
168 settings.rotation = LV_DISPLAY_ROTATION_90;
169 break;
170 case 180:
171 settings.rotation = LV_DISPLAY_ROTATION_180;
172 break;
173 case 270:
174 settings.rotation = LV_DISPLAY_ROTATION_270;
175 break;
176 default:
177 LV_LOG_WARN("Invalid rotation angle. Valid angles are {0, 90, 180, 270}");
178 break;
179 }
180 break;
181 case ':':
182 print_usage();
183 die("Option -%c requires an argument.\n", optopt);
184 break;
185 case '?':
186 print_usage();
187 die("Unknown option -%c.\n", optopt);
188 }
189 }
190}

206int main(int argc, char ** argv)
207{
208
209 configure_simulator(argc, argv);
210
211 /* Initialize LVGL. */
212 lv_init();
213
214 /* Initialize the configured default backend */
215 driver_backends_register();
216
217
218 /* Initialize the configured backend */
219 if(driver_backends_init_backend(selected_backend) == -1) {
220 die("Failed to initialize display backend");
221 }
222
223 if(settings.rotation) {
224#if LV_USE_DRAW_NANOVG && LV_DRAW_TRANSFORM_USE_MATRIX
225 lv_display_set_matrix_rotation(NULL, true);
226#endif
227 lv_display_set_rotation(NULL, settings.rotation);
228 }
229
230 /* Enable for EVDEV support */
231#if LV_USE_EVDEV
232 // if(driver_backends_init_backend("EVDEV") == -1) {
233 // die("Failed to initialize evdev");
234 // }
235#endif
236
237 /*Create a UI using .c file*/
238 // lv_ui_example();
239
240
241 printf("UI INIT !!!\n");
242 fflush(stdout);
243 /*Create a UI using generated files by eez-studio*/
244 ui_init();
245
246 if(app_runtime_init() != 0) {
247 die("Failed to initialize app runtime");
248 }
249
250 if(app_runtime_start() != 0) {
251 die("Failed to start app runtime");
252 }
253
254 /* Let LVGL drive the generated UI tick inside the active backend loop. */
255 lv_timer_create(ui_tick_timer_cb, 10, NULL);
256 lv_timer_create(ui_runtime_sync_timer_cb, 50, NULL);
257
258 /* Enter the run loop of the selected backend */
259 driver_backends_run_loop();
260
261 app_runtime_stop();
262
263 return 0;
264}
